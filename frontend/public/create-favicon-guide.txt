const scriptContent = `
const sharp = require('sharp');
const fs = require('fs');

// Create different sizes of favicon from a source image
async function createFavicons() {
  const sourceImage = 'logo.png'; // Your source image file
  
  if (!fs.existsSync(sourceImage)) {
    console.log('Source image not found. Creating a simple placeholder...');
    // Create a simple placeholder image with a shopping cart icon
    const canvas = document.createElement('canvas');
    canvas.width = 32;
    canvas.height = 32;
    const ctx = canvas.getContext('2d');
    
    // Draw a simple shopping cart icon
    ctx.fillStyle = '#667eea';
    ctx.fillRect(6, 10, 20, 12); // Cart body
    ctx.fillRect(8, 8, 2, 4);   // Handle
    ctx.fillRect(4, 22, 6, 4);  // Wheel 1
    ctx.fillRect(18, 22, 6, 4); // Wheel 2
    
    // Convert to buffer and save
    const imageData = canvas.toDataURL('image/png');
    // This approach won't work in Node.js environment
  } else {
    // Create different sizes
    await sharp(sourceImage)
      .resize(16, 16)
      .toFile('public/favicon-16x16.png');
      
    await sharp(sourceImage)
      .resize(32, 32)
      .toFile('public/favicon-32x32.png');
      
    // Create ICO file from PNG
    await sharp(sourceImage)
      .resize(32, 32)
      .toFile('public/favicon.ico');
    
    console.log('Favicons created successfully!');
  }
}

createFavicons().catch(console.error);
`;
